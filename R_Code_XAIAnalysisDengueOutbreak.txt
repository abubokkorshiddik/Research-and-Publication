# Load required libraries
library(readxl)       # For reading Excel files
library(tidyverse)    # For data manipulation
library(caret)        # For machine learning
library(randomForest) # For Random Forest
library(pROC)         # For ROC and AUC
library(openxlsx)     # For saving to Excel

# Load dataset
mydata <- read_excel("C:/R/LABSTAT/AIDengue20002023.xlsx", sheet = "Data")
print(head(mydata))  # View first few rows

# Filter data for years 2000–2019
data_2000_2019 <- mydata %>% filter(Year >= 2000 & Year <= 2019)

# Ensure `Cases_Bi` is treated as a factor for classification
data_2000_2019$Cases_Bi <- factor(data_2000_2019$Cases_Bi, levels = c(0, 1), labels = c("No", "Yes"))

# Create random subsets (10 folds)
set.seed(123)  # For reproducibility
folds <- createFolds(data_2000_2019$Cases_Bi, k = 10, list = TRUE, returnTrain = FALSE)

# Initialize an empty dataframe to store test set results
test_results <- data.frame(
  Fold = integer(),
  AUC = numeric(),
  Accuracy = numeric(),
  Sensitivity = numeric(),
  Specificity = numeric(),
  Precision = numeric(),
  F1 = numeric()
)

# Initialize a dataframe to store parameter tuning results (example for Random Forest)
parameter_results <- data.frame(
  Fold = integer(),
  mtry = integer()  # Random Forest hyperparameter
)

# Initialize a dataframe for storing hyperparameter tuning results
tuning_summary <- data.frame(
  Parameter = character(),
  Best_Value = character()
)

# Loop through each fold for train-test split
for (i in 1:10) {
  # Select test indices and split data
  test_indices <- folds[[i]]
  testData <- data_2000_2019[test_indices, ]
  trainData <- data_2000_2019[-test_indices, ]
  
  # Train a Random Forest model with all predictors (mtry = 25)
  set.seed(123)
  rf_model <- train(
    Cases_Bi ~ .,                       # Formula: Predict Cases_Bi
    data = trainData,                   # Training data
    method = "rf",                      # Random Forest
    metric = "ROC",                     # Optimize for AUC
    tuneGrid = data.frame(mtry = 25),   # Use all predictors (x1 to x25)
    trControl = trainControl(
      method = "none",                  # No resampling
      classProbs = TRUE,                # Enable probabilities
      summaryFunction = twoClassSummary # Evaluate ROC, Sensitivity, Specificity
    )
  )
  
  # Record hyperparameters for this fold
  parameter_results <- rbind(parameter_results, data.frame(
    Fold = i,
    mtry = 25  # Fixed value for mtry
  ))
  
  # Record parameter tuning results (only 1 value since no tuning is applied)
  tuning_summary <- rbind(tuning_summary, data.frame(
    Parameter = "mtry",
    Best_Value = 25
  ))
  
  # Make predictions on the test set
  rf_pred <- predict(rf_model, testData)                # Class predictions
  rf_prob <- predict(rf_model, testData, type = "prob")[, "Yes"]  # Probabilities for "Yes"
  
  # Evaluate metrics for the test set
  confusion <- confusionMatrix(rf_pred, testData$Cases_Bi)
  roc_curve <- roc(response = testData$Cases_Bi, predictor = rf_prob, levels = c("No", "Yes"))
  auc <- auc(roc_curve)
  
  # Add results for the current fold
  test_results <- rbind(test_results, data.frame(
    Fold = i,
    AUC = round(auc, 3),  # Rounded to 3 decimal places
    Accuracy = round(confusion$overall["Accuracy"], 3),
    Sensitivity = round(confusion$byClass["Sensitivity"], 3),
    Specificity = round(confusion$byClass["Specificity"], 3),
    Precision = round(confusion$byClass["Precision"], 3),
    F1 = round(confusion$byClass["F1"], 3)
  ))
}

# Display test set results for all 10 folds
print(test_results)

# Calculate average metrics
avg_metrics <- test_results %>%
  summarise(
    Avg_AUC = mean(AUC),
    Avg_Accuracy = mean(Accuracy),
    Avg_Sensitivity = mean(Sensitivity),
    Avg_Specificity = mean(Specificity),
    Avg_Precision = mean(Precision),
    Avg_F1 = mean(F1)
  ) %>%
  mutate(across(everything(), ~ round(.x, 3)))  # Round to 3 decimal places

print(avg_metrics)

# Save results to an Excel file with three sheets
write.xlsx(
  list(
    Fold_Results = test_results,      # Results for each fold
    Average_Metrics = avg_metrics,   # Average performance metrics
    Parameter_Tuning = tuning_summary # Hyperparameter tuning results
  ), 
  "RandomForest_Results.xlsx", 
  rowNames = FALSE
)

cat("Results have been saved to 'RandomForest_Results.xlsx'.\n")









# Load required libraries
library(readxl)
library(tidyverse)
library(caret)
library(xgboost)
library(pROC)
library(openxlsx)

# Load dataset
mydata <- read_excel("C:/R/LABSTAT/AIDengue20002023.xlsx", sheet = "Data")
print(head(mydata))

# Filter data for years 2000–2019
data_2000_2019 <- mydata %>% filter(Year >= 2000 & Year <= 2019)

# Ensure `Cases_Bi` is treated as a factor for classification
data_2000_2019$Cases_Bi <- factor(data_2000_2019$Cases_Bi, levels = c(0, 1), labels = c("No", "Yes"))

# Create random subsets (10 folds)
set.seed(123)
folds <- createFolds(data_2000_2019$Cases_Bi, k = 10, list = TRUE, returnTrain = FALSE)

# Initialize an empty dataframe to store test set results
test_results <- data.frame(
  Fold = integer(),
  AUC = numeric(),
  Accuracy = numeric(),
  Sensitivity = numeric(),
  Specificity = numeric(),
  Precision = numeric(),
  F1 = numeric()
)

# Initialize a dataframe to store the best hyperparameters from tuning
tuning_results <- data.frame(
  Fold = integer(),
  nrounds = integer(),
  max_depth = integer(),
  eta = numeric(),
  gamma = numeric(),
  colsample_bytree = numeric(),
  subsample = numeric(),
  min_child_weight = integer(),
  AUC = numeric()
)

# Loop through each fold for train-test split and hyperparameter tuning
for (i in 1:10) {
  # Select test indices and split data
  test_indices <- folds[[i]]
  testData <- data_2000_2019[test_indices, ]
  trainData <- data_2000_2019[-test_indices, ]
  
  # Convert predictors to matrix format required by XGBoost
  train_x <- model.matrix(~ . - 1, data = trainData %>% select(-Cases_Bi, -Year))
  test_x <- model.matrix(~ . - 1, data = testData %>% select(-Cases_Bi, -Year))
  
  # Align columns between train_x and test_x
  common_columns <- intersect(colnames(train_x), colnames(test_x))
  train_x <- train_x[, common_columns, drop = FALSE]
  test_x <- test_x[, common_columns, drop = FALSE]
  
  # Convert target variable to numeric (0 and 1) for XGBoost
  train_y <- as.numeric(trainData$Cases_Bi) - 1
  test_y <- as.numeric(testData$Cases_Bi) - 1
  
  # Define a smaller parameter grid
  param_grid <- expand.grid(
    nrounds = c(50, 100),         # Reduce boosting rounds
    max_depth = c(3, 4),          # Reduce depth values
    eta = c(0.1, 0.2),            # Reduce learning rates
    gamma = c(0),                 # Keep gamma minimal
    colsample_bytree = c(0.8),    # Simplify column sampling
    subsample = c(0.8),           # Simplify row sampling
    min_child_weight = c(1)       # Fixed child weight
  )
  
  # Use fewer cross-validation folds and switch to random search
  trControl <- trainControl(
    method = "random",            # Random search instead of grid search
    number = 10,                  # Test only 10 random combinations
    classProbs = TRUE,
    summaryFunction = twoClassSummary
  )
  
  # Perform hyperparameter tuning using caret
  set.seed(123)
  xgb_tuned <- train(
    x = train_x, 
    y = factor(train_y, levels = c(0, 1), labels = c("No", "Yes")),
    method = "xgbTree",
    tuneGrid = param_grid,
    metric = "ROC",
    trControl = trainControl(
      method = "cv",               # 5-fold cross-validation for tuning
      number = 5,
      classProbs = TRUE,
      summaryFunction = twoClassSummary
    )
  )
  
  # Store the best hyperparameters and AUC
  best_params <- xgb_tuned$bestTune
  tuning_results <- rbind(tuning_results, data.frame(
    Fold = i,
    nrounds = best_params$nrounds,
    max_depth = best_params$max_depth,
    eta = best_params$eta,
    gamma = best_params$gamma,
    colsample_bytree = best_params$colsample_bytree,
    subsample = best_params$subsample,
    min_child_weight = best_params$min_child_weight,
    AUC = max(xgb_tuned$results$ROC)
  ))
  
  # Make predictions using the tuned model
  xgb_pred_prob <- predict(xgb_tuned, newdata = test_x, type = "prob")[, "Yes"]
  xgb_pred_class <- predict(xgb_tuned, newdata = test_x)
  
  # Evaluate metrics for the test set
  confusion <- confusionMatrix(
    factor(xgb_pred_class, levels = c("No", "Yes")),
    factor(testData$Cases_Bi, levels = c("No", "Yes"))
  )
  roc_curve <- roc(response = testData$Cases_Bi, predictor = xgb_pred_prob, levels = c("No", "Yes"))
  auc <- auc(roc_curve)
  
  # Add results for the current fold
  test_results <- rbind(test_results, data.frame(
    Fold = i,
    AUC = round(auc, 3),
    Accuracy = round(confusion$overall["Accuracy"], 3),
    Sensitivity = round(confusion$byClass["Sensitivity"], 3),
    Specificity = round(confusion$byClass["Specificity"], 3),
    Precision = round(confusion$byClass["Precision"], 3),
    F1 = round(confusion$byClass["F1"], 3)
  ))
}

# Display test set results for all 10 folds
print(test_results)

# Display tuning results for all folds
print(tuning_results)

# Calculate average metrics across all folds
avg_metrics <- test_results %>%
  summarise(
    Avg_AUC = round(mean(AUC, na.rm = TRUE), 3),
    Avg_Accuracy = round(mean(Accuracy, na.rm = TRUE), 3),
    Avg_Sensitivity = round(mean(Sensitivity, na.rm = TRUE), 3),
    Avg_Specificity = round(mean(Specificity, na.rm = TRUE), 3),
    Avg_Precision = round(mean(Precision, na.rm = TRUE), 3),
    Avg_F1 = round(mean(F1, na.rm = TRUE), 3)
  )

# Save results to an Excel file with three sheets
write.xlsx(
  list(
    Fold_Results = test_results,
    Tuning_Results = tuning_results,
    Average_Metrics = avg_metrics
  ), 
  "XGBoost_Tuned_Results.xlsx", 
  rowNames = FALSE
)

cat("Results have been saved to 'XGBoost_Tuned_Results.xlsx'.\n")

  test_x <- testData %>% select(-Cases_Bi, -Year) %>% as.matrix()

  dtrain <- lgb.Dataset(data = train_x, label = train_y, free_raw_data = FALSE)

  param_grid <- expand.grid(

# Load required libraries
library(xgboost)
library(ROCR)
library(readxl)
library(openxlsx)

# Load dataset
mydata <- read_excel("C:/R/LABSTAT/AIDengue20002023.xlsx", sheet = "Data")

# Split data into training (2000-2019) and testing (2020-2023)
train <- subset(mydata, Year >= 2000 & Year <= 2019)
test <- as.data.frame(subset(mydata, Year >= 2020 & Year <= 2023))

# Ensure `Cases_Bi` column is binary and numeric for classification
train$Cases_Bi <- gsub("[^0-9]", "", train$Cases_Bi)  # Remove non-numeric characters, if any
train$CASE_BI <- as.numeric(train$Cases_Bi)           # Convert to numeric
test$Cases_Bi <- gsub("[^0-9]", "", test$Cases_Bi)
test$CASE_BI <- as.numeric(test$Cases_Bi)

# Remove rows with missing values
train <- na.omit(train)
test <- na.omit(test)

# Prepare feature matrices and labels
train_features <- as.matrix(train[, c("x1", "x2", "x3", "x4", "x5", "x6", "x7",
                                      "x8", "x9", "x10", "x11", "x12", "x13", "x14", "x15",
                                      "x16", "x17", "x18", "x19", "x20", "x21", "x22",
                                      "x23", "x24", "x25")])
train_labels <- train$CASE_BI
test_features <- as.matrix(test[, c("x1", "x2", "x3", "x4", "x5", "x6", "x7",
                                    "x8", "x9", "x10", "x11", "x12", "x13", "x14", "x15",
                                    "x16", "x17", "x18", "x19", "x20", "x21", "x22",
                                    "x23", "x24", "x25")])
test_labels <- test$CASE_BI

# Hyperparameter tuning grid
param_grid <- expand.grid(
  eta = c(0.01, 0.1, 0.3),         # Learning rate
  max_depth = c(3, 6, 10),         # Maximum tree depth
  nrounds = c(50, 100, 150),       # Boosting iterations
  gamma = c(0, 1),                 # Minimum loss reduction to split
  colsample_bytree = c(0.6, 0.8),  # Fraction of columns used per tree
  min_child_weight = c(1, 5),      # Minimum child weight
  subsample = c(0.6, 0.8)          # Fraction of observations sampled per tree
)

# Track the best model and hyperparameters
best_auc <- 0
best_params <- list()

# Loop through the parameter grid to train and evaluate models
for (i in 1:nrow(param_grid)) {
  params <- list(
    eta = param_grid$eta[i],
    max_depth = param_grid$max_depth[i],
    gamma = param_grid$gamma[i],
    colsample_bytree = param_grid$colsample_bytree[i],
    min_child_weight = param_grid$min_child_weight[i],
    subsample = param_grid$subsample[i],
    objective = "binary:logistic",  # Binary classification
    eval_metric = "auc"            # Use AUC as the evaluation metric
  )
  
  # Train XGBoost model
  model <- xgboost(
    params = params,
    data = train_features,
    label = train_labels,
    nrounds = param_grid$nrounds[i],
    verbose = 0
  )
  
  # Predict on the test set
  pred <- predict(model, test_features)
  
  # Calculate AUC
  pred_obj <- prediction(pred, test_labels)
  auc <- performance(pred_obj, "auc")@y.values[[1]]
  
  # Update the best model if necessary
  if (auc > best_auc) {
    best_auc <- auc
    best_params <- params
    best_params$nrounds <- param_grid$nrounds[i]
  }
}

# Train final model with best hyperparameters
final_model <- xgboost(
  params = best_params,
  data = train_features,
  label = train_labels,
  nrounds = best_params$nrounds,
  verbose = 0
)

# Make final predictions on the test set
final_pred <- predict(final_model, test_features)

# Generate Confusion Matrix
final_pred_class <- ifelse(final_pred > 0.5, 1, 0)
conf_matrix <- table(Predicted = final_pred_class, Actual = test_labels)

# Calculate Metrics
accuracy <- sum(diag(conf_matrix)) / sum(conf_matrix)
precision <- conf_matrix[2, 2] / sum(conf_matrix[2, ])
recall <- conf_matrix[2, 2] / sum(conf_matrix[, 2])
specificity <- conf_matrix[1, 1] / sum(conf_matrix[, 1])
f1 <- (2 * precision * recall) / (precision + recall)

# Output Metrics
cat("Performance Metrics:\n")
cat("Best AUC:", best_auc, "\n")
cat("Accuracy:", accuracy, "\n")
cat("Precision:", precision, "\n")
cat("Recall (Sensitivity):", recall, "\n")
cat("Specificity:", specificity, "\n")
cat("F1-Score:", f1, "\n")

# Save results to Excel
results <- data.frame(
  Metric = c("Best AUC", "Accuracy", "Precision", "Recall (Sensitivity)", "Specificity", "F1-Score"),
  Value = c(best_auc, accuracy, precision, recall, specificity, f1)
)

# Load required libraries
library(readxl)
library(dplyr)
library(openxlsx)

# Load the dataset
mydata <- read_excel("C:/R/LABSTAT/AIDengue20002023.xlsx", sheet = "Data")

# Prepare the dataset: order months correctly
mydata <- mydata %>%
  mutate(Months1 = factor(Months1, levels = c("Jan", "Feb", "Mar", "Apr", "May", "Jun",
                                              "Jul", "Aug", "Sep", "Oct", "Nov", "Dec")))

# Calculate yearly totals for cases and deaths
yearly_totals <- mydata %>%
  group_by(Year) %>%
  summarise(
    Total_Cases = sum(Cases, na.rm = TRUE),
    Total_Deaths = sum(Deaths, na.rm = TRUE)
  )

# 2 highest months with cases and deaths for each year
highest_months <- mydata %>%
  group_by(Year) %>%
  arrange(desc(Cases)) %>%
  filter(row_number() <= 2) %>%  # Keep top 2 months
  select(Year, Months1, Cases, Deaths)

# 2 lowest months with cases and deaths for each year
lowest_months <- mydata %>%
  group_by(Year) %>%
  arrange(Cases) %>%
  filter(row_number() <= 2) %>%  # Keep bottom 2 months
  select(Year, Months1, Cases, Deaths)

# Year with the highest total cases and deaths
highest_total_cases <- yearly_totals[which.max(yearly_totals$Total_Cases), ]
highest_total_deaths <- yearly_totals[which.max(yearly_totals$Total_Deaths), ]

# Year with the lowest total cases and deaths
lowest_total_cases <- yearly_totals[which.min(yearly_totals$Total_Cases), ]
lowest_total_deaths <- yearly_totals[which.min(yearly_totals$Total_Deaths), ]

# Save the results to an Excel file
wb <- createWorkbook()

# Add yearly totals to Sheet 1
addWorksheet(wb, "Yearly Totals")
writeData(wb, "Yearly Totals", yearly_totals)

# Add 2 highest months per year to Sheet 2
addWorksheet(wb, "Highest Months")
writeData(wb, "Highest Months", highest_months)

# Add 2 lowest months per year to Sheet 3
addWorksheet(wb, "Lowest Months")
writeData(wb, "Lowest Months", lowest_months)

# Add highest and lowest yearly totals to Sheet 4
addWorksheet(wb, "Highest & Lowest Totals")
highest_lowest <- data.frame(
  Description = c("Year with Highest Cases", "Year with Lowest Cases", 
                  "Year with Highest Deaths", "Year with Lowest Deaths"),
  Year = c(highest_total_cases$Year, lowest_total_cases$Year, 
           highest_total_deaths$Year, lowest_total_deaths$Year),
  Total_Cases = c(highest_total_cases$Total_Cases, lowest_total_cases$Total_Cases,
                  highest_total_deaths$Total_Cases, lowest_total_deaths$Total_Cases),
  Total_Deaths = c(highest_total_cases$Total_Deaths, lowest_total_cases$Total_Deaths,
                   highest_total_deaths$Total_Deaths, lowest_total_deaths$Total_Deaths)
)
writeData(wb, "Highest & Lowest Totals", highest_lowest)

# Save the workbook
saveWorkbook(wb, file = "Y1early_Summary_Analysis.xlsx", overwrite = TRUE)

cat("Yearly summary with 2 highest and 2 lowest months saved in 'Yearly_Summary_Analysis.xlsx'.")

# Step 1: Load and Preprocess the Data
mydata <- read_excel("C:/R/LABSTAT/AIDengue20002023.xlsx", sheet = "Data")

# Clean target variables and exclude unnecessary columns
mydata <- mydata %>%
  mutate(Cases_Bi1 = ifelse(is.na(Cases_Bi1) | is.infinite(Cases_Bi1) | Cases_Bi1 > 1, 0, Cases_Bi1),
         Cases_Bi = ifelse(is.na(Cases_Bi) | is.infinite(Cases_Bi) | Cases_Bi > 1, 0, Cases_Bi))

excluded_columns <- c("Year", "Months", "Months1", "Cases", "Deaths", "Cases_Bi", "Cases_Bi1")

# Split the data into training (2000-2019) and testing (2020-2023)
train_data <- filter(mydata, Year >= 2000 & Year <= 2019)
test_data <- filter(mydata, Year >= 2020 & Year <= 2023)

# Prepare features and target variables
X_train <- select(train_data, -all_of(excluded_columns))
X_test <- select(test_data, -all_of(excluded_columns))
y_train_case_bi <- train_data$Cases_Bi
y_test_case_bi <- test_data$Cases_Bi
y_train_case_bi1 <- train_data$Cases_Bi1
y_test_case_bi1 <- test_data$Cases_Bi1

# Step 2: Train XGBoost Models
xgb_model_case_bi <- xgboost(
  data = as.matrix(X_train), label = y_train_case_bi,
  nrounds = 100, objective = "binary:logistic", eval_metric = "logloss"
)

xgb_model_case_bi1 <- xgboost(
  data = as.matrix(X_train), label = y_train_case_bi1,
  nrounds = 100, objective = "binary:logistic", eval_metric = "logloss"
)

# Step 3: Extract Feature Importance
# Feature importance for Case_Bi
importance_case_bi <- xgb.importance(feature_names = colnames(X_train), model = xgb_model_case_bi)
print("Top 10 Features for Case_Bi:")
print(importance_case_bi[1:10, ])

# Feature importance for Case_Bi1
importance_case_bi1 <- xgb.importance(feature_names = colnames(X_train), model = xgb_model_case_bi1)
print("Top 10 Features for Case_Bi1:")
print(importance_case_bi1[1:10, ])

# Plot feature importance for Case_Bi
xgb.plot.importance(importance_case_bi, top_n = 10, main = "Top 10 Features for Case_Bi")

# Plot feature importance for Case_Bi1
xgb.plot.importance(importance_case_bi1, top_n = 10, main = "Top 10 Features for Case_Bi1")

# Step 4: Explain Models with DALEX
# Create DALEX explainers
explainer_case_bi <- explain(
  model = xgb_model_case_bi,
  data = X_test,
  y = y_test_case_bi,
  label = "Case_Bi Model",
  predict_function = function(model, newdata) {
    predict(model, as.matrix(newdata))
  }
)

explainer_case_bi1 <- explain(
  model = xgb_model_case_bi1,
  data = X_test,
  y = y_test_case_bi1,
  label = "Case_Bi1 Model",
  predict_function = function(model, newdata) {
    predict(model, as.matrix(newdata))
  }
)

# Generate and Plot Variable Importance
vip_case_bi <- model_parts(explainer_case_bi)
plot(vip_case_bi, title = "Variable Importance for Case_Bi")

  "PopDen", "GDP", "PovertyHC", "Cases", "Deaths"
)
mydata_cleaned <- na.omit(mydata[, relevant_features])

# Step 3: Prepare Training Data for Cases and Deaths
train_matrix_cases <- xgb.DMatrix(
  data = as.matrix(mydata_cleaned[, -which(names(mydata_cleaned) %in% c("Cases", "Deaths"))]),
  label = mydata_cleaned$Cases
)

train_matrix_deaths <- xgb.DMatrix(
  data = as.matrix(mydata_cleaned[, -which(names(mydata_cleaned) %in% c("Cases", "Deaths"))]),
  label = mydata_cleaned$Deaths
)

# Step 4: Train XGBoost Models for Cases and Deaths
xgb_params <- list(
  objective = "reg:squarederror",
  eval_metric = "rmse",
  eta = 0.1,
  max_depth = 6
)
xgb_model_cases <- xgb.train(params = xgb_params, data = train_matrix_cases, nrounds = 100)
xgb_model_deaths <- xgb.train(params = xgb_params, data = train_matrix_deaths, nrounds = 100)

# Step 5: Generate Future Data (2024–2031)
future_years <- data.frame(
  Year = rep(2024:2031, each = 12),
  Month = rep(1:12, times = 8)
)

# Add realistic seasonal patterns and trends
monthly_mean_tem <- tapply(mydata$MeanTem, mydata$Months, mean, na.rm = TRUE)
monthly_rainfall <- tapply(mydata$Rainfall, mydata$Months, mean, na.rm = TRUE)
gdp_trend <- lm(GDP ~ Year, data = mydata)
poverty_trend <- lm(PovertyHC ~ Year, data = mydata)

future_years$MeanTem <- rep(monthly_mean_tem, times = 8)
future_years$Rainfall <- rep(monthly_rainfall, times = 8)
future_years$GDP <- predict(gdp_trend, newdata = data.frame(Year = future_years$Year))
future_years$PovertyHC <- predict(poverty_trend, newdata = data.frame(Year = future_years$Year))
future_years$PopDen <- rep(mean(mydata$PopDen, na.rm = TRUE), times = nrow(future_years))

# Align columns explicitly
required_columns <- colnames(mydata_cleaned[, -which(names(mydata_cleaned) %in% c("Cases", "Deaths"))])
missing_columns <- setdiff(required_columns, colnames(future_years))

for (col in missing_columns) {
  future_years[[col]] <- mean(mydata[[col]], na.rm = TRUE)
}
future_years <- future_years[, required_columns]

# Step 6: Prepare Future Data Matrix
future_matrix <- xgb.DMatrix(data = as.matrix(future_years))

# Step 7: Predict Cases and Deaths for Each Month

future_years$PredictedDeaths <- predict(xgb_model_deaths, future_matrix)

# Function to calculate MAE
calculate_mae <- function(actual, predicted) {
  mean(abs(actual - predicted))
}

calculate_mape <- function(actual, predicted) {
  # Exclude rows where actual == 0 to avoid division by zero
  valid_indices <- actual != 0
  mean(abs((actual[valid_indices] - predicted[valid_indices]) / actual[valid_indices])) * 100
}

  pdp_ranges_xgb <- rbind(pdp_ranges_xgb, range_data)
}

# Step 4: Save PDP Ranges to Excel
write_xlsx(list("XGBoost PDP Ranges for Deaths" = pdp_ranges_xgb), path = "XGBoost_Deaths_PDP_Ranges.xlsx")

# Step 5: Generate PDP Plots for XGBoost (Deaths)
plot_titles_xgb <- LETTERS[1:25]  # Titles A, B, C, etc.

for (i in seq_along(top_features_xgb)) {
  feature <- top_features_xgb[i]
  pd <- partial(xgb_model, pred.var = feature, train = X_train, grid.resolution = 50)
  pd_plot <- autoplot(pd) +
    labs(title = paste(plot_titles_xgb[i]), x = feature, y = "Predicted Deaths") +
    theme_minimal()
  
  # Save each plot separately
  ggsave(filename = paste0(plot_titles_xgb[i], "_PDP_XGBoost_Deaths_", feature, ".png"), 
         plot = pd_plot, width = 6, height = 4)
}
