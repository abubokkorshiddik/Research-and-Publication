prepare_map <- function(year_sel) {

# 4. Create map function with DENV points (color only, jittered)
create_map_with_DENV <- function(mapdata, yr, brks) {
  ggplot(mapdata) +
    # Polygon fill for yearly cases
    geom_sf(aes(fill = Yearly_cases), color = "black", size = 0.15) +
    
    # Points for DENV1–4 if > 0 (color indicates type, jittered)
    geom_point(data = subset(mapdata, DENV1 > 0), 
               aes(x = Longitude, y = Latitude, color = "DENV1"),

               size = 2, position = position_jitter(width = 0.2, height = 0.2)) +

               aes(x = Longitude, y = Latitude, color = "DENV3"),
               size = 2, position = position_jitter(width = 0.2, height = 0.2)) +
    geom_point(data = subset(mapdata, DENV4 > 0), 
               aes(x = Longitude, y = Latitude, color = "DENV4"),
               size = 2, position = position_jitter(width = 0.2, height = 0.2)) +
    
    # Fill scale for yearly cases
    scale_fill_gradientn(
      colours = c("white","steelblue", "skyblue", "orange", "red", "darkred"),
      values = scales::rescale(brks),
      breaks = brks,
      labels = brks,
      name = "Cases"

      values = c("DENV1" = "darkmagenta", "DENV2" = "darkgreen",

    # North arrow and scale
    annotation_scale(location = "bl", width_hint = 0.3) +
    annotation_north_arrow(location = "tr", which_north = "true",
                           style = north_arrow_nautical()) +
    
    # Labels and theme
    labs(title = yr, x = NULL, y = NULL) +
    theme_minimal(base_size = 14) +
    theme(
      plot.title = element_text(face = "bold", hjust = 0.5),
      legend.position = "right",
      axis.text = element_blank(),    # remove tick labels
      axis.title = element_blank(),   # remove axis names
      axis.ticks = element_blank(),   # remove tick marks

      panel.border = element_rect(color = "black", fill = NA, size = 0.5),
      text = element_text(face = "bold")
    )

# ============================================================
# 4. Compute Getis-Ord Gi* for a given year with CI categories
# ============================================================
compute_gistar_ci <- function(year_sel) {
  
  mapdata <- districts %>%
    left_join(mydata %>% filter(Year == year_sel) %>% 
                select(District, Yearly_cases),
              by = c("ADM2_EN" = "District"))
  
  mapdata$Yearly_cases[is.na(mapdata$Yearly_cases)] <- 0
  
  nb <- poly2nb(mapdata, queen = TRUE)
  lw <- nb2listw(nb, style = "W", zero.policy = TRUE)
  
  gi <- localG(mapdata$Yearly_cases, lw, zero.policy = TRUE)
  mapdata$GiZ <- as.numeric(gi)
  
  # Classification by CI
  mapdata$HotCold <- cut(
    mapdata$GiZ,
    breaks = c(-Inf, -2.576, -1.96, -1.645, 1.645, 1.96, 2.576, Inf),
    labels = c(
      "Cold spot - 99% CI",
      "Cold spot - 95% CI",
      "Cold spot - 90% CI",
      "Not significant",
      "Hot spot - 90% CI",
      "Hot spot - 95% CI",
      "Hot spot - 99% CI"
    ),
    include.lowest = TRUE
  )
  
  return(mapdata)
}

# ============================================================
# 4. Function to compute Local Moran's I (LISA)
# ============================================================
compute_lisa <- function(year_sel) {
  
  mapdata <- districts %>%
    left_join(mydata %>% filter(Year == year_sel) %>% select(District, Yearly_cases),
              by = c("ADM2_EN" = "District"))
  
  mapdata$Yearly_cases[is.na(mapdata$Yearly_cases)] <- 0
  
  nb <- poly2nb(mapdata, queen = TRUE)
  lw <- nb2listw(nb, style = "W", zero.policy = TRUE)
  
  lisa <- localmoran(mapdata$Yearly_cases, lw, zero.policy = TRUE)
  
  # z-score of local Moran's I
  mapdata$LisaZ <- (lisa[,1] - mean(lisa[,1])) / sd(lisa[,1])
  
  # Classify clusters using z-score thresholds
  mapdata$LisaCat <- cut(
    mapdata$LisaZ,
    breaks = c(-Inf, -2.576, -1.96, -1.645, 1.645, 1.96, 2.576, Inf),
    labels = c(
      "Low-Low 99%", "Low-Low 95%", "Low-Low 90%",
      "Not Significant",
      "High-High 90%", "High-High 95%", "High-High 99%"
    ),
    include.lowest = TRUE
  )
  
  return(mapdata)
}

# ============================================================
# 3. Function to Generate Summary for Each DENV
# ============================================================
generate_summary <- function(df, denv_col, yearly_total_df) {
  
  summary_df <- df %>%
    group_by(Year) %>%
    summarise(
      Min = round(min(.data[[denv_col]], na.rm = TRUE)),
      Q1 = round(quantile(.data[[denv_col]], 0.25, na.rm = TRUE)),
      Median = round(median(.data[[denv_col]], na.rm = TRUE)),
      Q3 = round(quantile(.data[[denv_col]], 0.75, na.rm = TRUE)),
      Max = round(max(.data[[denv_col]], na.rm = TRUE)),
      Mean = round(mean(.data[[denv_col]], na.rm = TRUE)),
      SD = round(sd(.data[[denv_col]], na.rm = TRUE)),
      Mean_SD = paste0(round(mean(.data[[denv_col]], na.rm = TRUE), 0),
                       " ± ",
                       round(sd(.data[[denv_col]], na.rm = TRUE), 0))
    ) %>%
    left_join(yearly_total_df, by = "Year") %>%
    mutate(
      Percentage = round((Mean / Total) * 100, 2)
    ) %>%
    select(-Total)
  
  return(summary_df)
}

shapefile_path <- "C:/R/LABSTAT/bgd_adm_bbs_20201113_SHP"
districts <- st_read(shapefile_path, layer = "bgd_admbnda_adm2_bbs_20201113")
districts$ADM2_EN <- toupper(trimws(districts$ADM2_EN))

# ============================================================
# 2. Load dengue dataset
# ============================================================
file_path <- "J:/Research/Research/denguebangladesh/data/ABSDengueDataBD.xlsx"
mydata <- readxl::read_excel(file_path, sheet = "data11")
mydata$District <- toupper(trimws(mydata$District))

# Keep only districts present in shapefile
mydata <- mydata %>% filter(District %in% districts$ADM2_EN)

# ============================================================
# 3. Create Spatial Weights (Queen contiguity)
# ============================================================
nb <- poly2nb(districts, queen = TRUE)
W_space <- nb2listw(nb, style = "W")
S <- length(nb)

# ============================================================
# 4. Create Temporal Weights (lag-1 adjacency)
# ============================================================
years <- sort(unique(mydata$Year))
T <- length(years)
W_time <- matrix(0, nrow = T, ncol = T)
for (i in 1:(T-1)) {
  W_time[i, i+1] <- 1
  W_time[i+1, i] <- 1
}
row_sums <- rowSums(W_time)
row_sums[row_sums == 0] <- 1
W_time <- W_time / row_sums

# ============================================================
# 5. Spatio-Temporal Weight Matrix
# ============================================================
W_ST <- kronecker(W_time, listw2mat(W_space))  # (S*T) x (S*T)

# ============================================================
# 6. Prepare panel data in district-year order
# ============================================================
district_order <- districts$ADM2_EN
year_order <- years

grid <- expand.grid(District = district_order, Year = year_order)

mydata_ordered <- grid %>%
  left_join(mydata, by = c("District", "Year"))

# Replace NA with 0
mydata_ordered[is.na(mydata_ordered)] <- 0

# ============================================================
# 7. Select numeric variables, exclude unwanted ones
# ============================================================
exclude_vars <- c("Latitude","Longitude","DENV1","DENV2","DENV3","DENV4",
                  "X17","X110","X119","X114","X115","X117","X113", "X18", "Year")

num_vars <- names(mydata_ordered)[sapply(mydata_ordered, is.numeric) & !(names(mydata_ordered) %in% exclude_vars)]
X <- as.matrix(mydata_ordered[, num_vars])

# Standardize predictors (z-score)
X_scaled <- scale(X)

# Identify Yearly_cases column
y_idx <- which(colnames(X_scaled) == "Yearly_cases")
yearly_cases <- X_scaled[, y_idx]

# ============================================================
# 8. Compute Spatio-Temporal Weighted Correlations
# ============================================================
results <- data.frame()

for (i in 1:ncol(X_scaled)) {
  if (i == y_idx) next
  
  x_var <- X_scaled[, i]
  y_var <- yearly_cases
  
  # Lagged values
  x_lag <- W_ST %*% x_var
  y_lag <- W_ST %*% y_var
  
  # Pearson correlation
  st_cor <- cor(y_lag, x_lag, use = "complete.obs")
  
  # p-value
  test <- cor.test(y_lag, x_lag, method = "pearson")
  p_val <- test$p.value
  
  # 95% CI using Fisher Z
  n_obs <- sum(!is.na(x_lag) & !is.na(y_lag))
  fisher_z <- 0.5 * log((1 + st_cor)/(1 - st_cor))
  se_z <- 1 / sqrt(n_obs - 3)
  z_crit <- qnorm(0.975)
  lower_z <- fisher_z - z_crit * se_z
  upper_z <- fisher_z + z_crit * se_z
  lower_r <- (exp(2*lower_z) - 1) / (exp(2*lower_z) + 1)
  upper_r <- (exp(2*upper_z) - 1) / (exp(2*upper_z) + 1)
  
  # Significance stars
  if (p_val < 0.01) {
    sig <- "**"
  } else if (p_val < 0.05) {
    sig <- "*"
  } else {
    sig <- ""
  }
  
  results <- rbind(results, data.frame(
    Variable = colnames(X_scaled)[i],
    ST_Correlation = round(st_cor, 3),
    CI_95 = paste0("(", round(lower_r,3), ", ", round(upper_r,3), ")"),
    P_value = round(p_val, 3),
    Significance = sig
  ))
}

# ============================================================
# 9. Format results for CSV (Correlation + 95% CI, P-value with stars)
# ============================================================
results_formatted <- results %>%
  mutate(
    Correlation_CI = paste0(ST_Correlation, " ", CI_95),
    P_value_star = ifelse(P_value < 0.01, "<0.01**",
                          ifelse(P_value < 0.05, "<0.05*", formatC(P_value, format = "f", digits = 3)))
  ) %>%
  select(Variable, Correlation_CI, P_value_star)

# ============================================================
# 10. Save formatted CSV
# ============================================================
output_csv_formatted <- "J:/Research/Research/denguebangladesh/yearlyST_correlation_results_formatted.csv"
write.csv(results_formatted, file = output_csv_formatted, row.names = FALSE)

# ============================================================
# 11. Plot significant variables
# ============================================================
sig_results <- results %>% filter(Significance != "")

sig_results <- sig_results %>%
  tidyr::separate(CI_95, into = c("Lower_CI", "Upper_CI"), sep = ", ", remove = FALSE) %>%
  mutate(
    Lower_CI = as.numeric(gsub("\\(|\\)", "", Lower_CI)),
    Upper_CI = as.numeric(gsub("\\)|\\(", "", Upper_CI))
  ) %>%
  arrange(ST_Correlation) %>%
  mutate(Variable = factor(Variable, levels = Variable))

# Plot
p <- ggplot(sig_results, aes(x = reorder(Variable, ST_Correlation), y = ST_Correlation, fill = Significance)) +
  geom_bar(stat = "identity") +
  geom_errorbar(aes(ymin = Lower_CI, ymax = Upper_CI), width = 0.3, color = "black") +
  coord_flip() +
  scale_fill_manual(values = c("*" = "skyblue", "**" = "steelblue")) +
  labs(
    title = "A",
    x = "Variables",
    y = "ST weighted correlation",
    fill = "Significance"
  ) +
  theme_minimal(base_size = 14) +
  theme(
    axis.text.y = element_text(face = "bold"),
    axis.title.y = element_text(face = "bold"),
    axis.title.x = element_text(face = "bold"),
    plot.title = element_text(face = "bold", hjust = 0),
    legend.position = "bottom",
    legend.title = element_text(face = "bold"),
    legend.text = element_text(face = "bold")
  )

# Show plot
print(p)

# Save plot
ggsave("J:/Research/Research/denguebangladesh/yearlystSignificant_Correlations.png",
       plot = p, width = 10, height = 5, dpi = 300)

cat("ST correlation analysis complete.\nFormatted CSV and plot saved.\n")

data$District <- factor(data$District, levels = districts_unique)

data$TimeID <- as.numeric(data$MonthYear)

# ============================================================
top10_vars <- c("X24","X216","X22","X28","X212","X25","X220","X21","X23")
data[top10_vars] <- scale(data[top10_vars])

# ============================================================
# 3. Helper function to create lagged responses
# ============================================================
lag_response <- function(df, lags = 1:3) {
  df <- df %>% arrange(District, Year, Month1)
  for(l in lags){
    df[[paste0("lag", l)]] <- ave(df$Monthly_cases, df$District, FUN = function(x) {
      if(length(x) <= l) return(rep(NA, length(x)))
      dplyr::lag(x, l)
    })
  }
  df <- df %>% filter(rowSums(is.na(df[paste0("lag", lags)])) == 0)
  return(df)
}

# ============================================================
# 4. Define lag options
# ============================================================
lag_options <- list(
  "Lag1"   = 1,
  "Lag12"  = 1:2,
  "Lag123" = 1:3
)

# ============================================================
# 5. Fit lagged models
# ============================================================
results_lagged <- list()
for(lag_name in names(lag_options)){
  
  df_lag <- lag_response(data, lags = lag_options[[lag_name]])
  
  if(nrow(df_lag) == 0){
    cat("Skipping", lag_name, "because dataset is empty after lagging\n")
    next
  }
  
  lag_terms <- paste0("lag", lag_options[[lag_name]], collapse = " + ")
  
  formula_lag <- as.formula(
    paste("Monthly_cases ~", paste(top10_vars, collapse = " + "), "+", lag_terms,
          "+ f(DistrictID, model='iid') + f(TimeID, model='rw1')")  # IID_RW1 base
  )
  
  cat("Fitting lagged model:", lag_name, "\n")
  
  results_lagged[[lag_name]] <- inla(
    formula_lag,
    data = df_lag,
    family = "nbinomial",
    control.compute = list(dic=TRUE, waic=TRUE, cpo=TRUE)
  )
}

# ============================================================
# 6. Summarize metrics
# ============================================================
lagged_metrics <- data.frame(
  Model = names(results_lagged),
  DIC = sapply(results_lagged, function(x) x$dic$dic),
  WAIC = sapply(results_lagged, function(x) x$waic$waic),
  meanCPO = sapply(results_lagged, function(x) mean(-log(x$cpo$cpo))),
  MargLogLik = sapply(results_lagged, function(x) x$mlik[1,1])
)

# 7. Create lagged dataset
# ============================================================
data <- data %>% arrange(District, Year, Month1) %>%
  group_by(District) %>%
  mutate(
    lag1 = dplyr::lag(Monthly_cases, 1, default = 0),
    lag2 = dplyr::lag(Monthly_cases, 2, default = 0),
    lag3 = dplyr::lag(Monthly_cases, 3, default = 0)
  ) %>% ungroup()

        "+ f(DistrictID, model='iid') + f(TimeID, model='rw1')")
)
