# Prepare the data: Remove rows with missing values in any of the selected columns
data_model <- data %>%
  select(x1, x2, x3, x4, y1, y6) %>%
  drop_na()  # Drop rows with any NA values

# Train a random forest model to predict x1 (malaria cases)

model <- randomForest(x1 ~ x2 + x3 + x4 + y1 + y6, data = data_model, importance = TRUE)

# View the model summary
print(model)

# Plot the variable importance to understand what influences x1 the most
importance(model)
varImpPlot(model)

# Predict malaria cases (x1) using the trained model
data_model$predicted_x1 <- predict(model, newdata = data_model)

# View the predicted values
head(data_model$predicted_x1)

# Visualize the actual vs predicted values
ggplot(data_model, aes(x = x1, y = predicted_x1)) +
  geom_point() +
  geom_smooth(method = "lm", se = FALSE, color = "blue") +
  labs(title = "Actual vs Predicted Malaria Cases (x1)",
       x = "Actual x1 (Malaria Cases)", y = "Predicted x1 (Malaria Cases)")



# Install and load the necessary libraries for DALEX
install.packages("DALEX")
library(DALEX)

# Assuming you already have your random forest model (from previous steps) as 'model'

# 1. Prepare the explainer object using DALEX
# We create an explainer to use the DALEX functions
explainer <- explain(model, 
                     data = data_model[, c("x2", "x3", "x4", "y1", "y6")], 
                     y = data_model$x1)

# 2. Global Interpretation: Feature Importance
# Calculate the global feature importance
global_importance <- model_parts(explainer)
plot(global_importance)

# 3. Local Interpretation: Single Prediction Explanation
# We want to explain the prediction for a specific instance, for example, the first row of data
single_instance <- data_model[1, c("x2", "x3", "x4", "y1", "y6")]
local_explanation <- predict_parts(explainer, new_observation = single_instance)
plot(local_explanation)

# 4. Partial Dependence Plot: Visualizing the effect of each feature
# For example, visualize the effect of 'x2' (malaria deaths) on the predicted malaria cases (x1)
pdp_x2 <- partial_dependence(explainer, variable = "x2")
plot(pdp_x2)

# You can similarly explore partial dependence plots for other features
pdp_y1 <- partial_dependence(explainer, variable = "y1")
plot(pdp_y1)

# Define Root Mean Squared Error (RMSE) function
rmse <- function(y_true, y_pred) {
  sqrt(mean((y_true - y_pred)^2))
}

# Define Mean Absolute Error (MAE) function
mae <- function(y_true, y_pred) {
  mean(abs(y_true - y_pred))
}

# Define Adjusted R2 function
adj_r2 <- function(model, X, y) {
  preds <- predict(model, X)
  r2 <- 1 - sum((y - preds)^2) / sum((y - mean(y))^2)
  n <- length(y)
  p <- ncol(X)
  adj_r2 <- 1 - ((1 - r2) * (n - 1)) / (n - p - 1)
  return(adj_r2)
}

### Multiple Regression Model ###
multi_formula <- as.formula(`Malaria incidence` ~ y1 + y2 + y3 + y4 + y5 + y6 + y7 + y8 + y9 + y10 + y11 + y12 + y13 + y14 + y15 + y16 + y17 + y18 + y19)
fit_multi <- lm(multi_formula, data = distrain)
preds_multi <- predict(fit_multi, newdata = distest)
rmse_multi <- rmse(distest$`Malaria incidence`, preds_multi)
r2_multi <- summary(fit_multi)$r.squared
mae_multi <- mae(distest$`Malaria incidence`, preds_multi)
adj_r2_multi <- 1 - ((1 - r2_multi) * (nrow(distest) - 1)) / (nrow(distest) - length(fit_multi$coefficients) - 1)

### Naive Model ###
preds_naive <- rep(mean(distrain$`Malaria incidence`), nrow(distest))

mae_naive <- mae(distest$`Malaria incidence`, preds_naive)
adj_r2_naive <- 1 - ((1 - r2_naive) * (nrow(distest) - 1)) / (nrow(distest) - 1 - 1)

# Initialize list to store results
early_warning_results <- list()

# Forecast years
future_years <- data.frame(Year = 2023:2040)

# Loop over each country
for (country in countries) {
  
  # Filter data for the specific country
  country_data <- mydata %>% filter(Country == country)
  
  # Select relevant columns (y1 to y22 + target variables)
  selected_vars <- c("Year", "Total_Incidence", "Total_Mortality", paste0("y", 1:22))
  data_train <- country_data %>% select(all_of(selected_vars))
  
  # Handle missing values by replacing them with 0
  data_train <- data_train %>%
    mutate(across(everything(), ~ replace_na(as.numeric(.), 0)))
  
  # Initialize forecast data
  forecast_data <- future_years
  
  # Trend-based extrapolation for y1 to y22 and target variables
  for (var in c("Total_Incidence", "Total_Mortality", paste0("y", 1:22))) {
    
    # Check if the variable has variation (not all zeros)
    if (sum(data_train[[var]]) > 0) {
      
      # Fit a linear model
      trend_model <- lm(as.formula(paste(var, "~ Year")), data = data_train)
      
      # Fit a polynomial model (degree 2) for non-linear trends
      poly_model <- lm(as.formula(paste(var, "~ poly(Year, 2)")), data = data_train)
      
      # Predict using both models
      linear_pred <- predict(trend_model, newdata = forecast_data)
      poly_pred <- predict(poly_model, newdata = forecast_data)
      
      # Use polynomial prediction if it gives more realistic values
      forecast_data[[var]] <- ifelse(poly_pred > 0, poly_pred, linear_pred)
      
    } else {
      # If all values are 0, set forecast to 0
      forecast_data[[var]] <- 0
    }
  }
  
  # Ensure no negative values
  forecast_data <- forecast_data %>%
    mutate(across(where(is.numeric), ~ pmax(., 0)))
  
  # Add country name
  forecast_data$Country <- country
  
  # Store results
  early_warning_results[[country]] <- forecast_data
}

# Load necessary libraries
library(caret)
library(xgboost)

# Set seed for reproducibility
set.seed(123)

# Define XGBoost parameters
params <- list(
  objective = "reg:squarederror", 
  eval_metric = "rmse",  
  max_depth = 6, 
  eta = 0.05,  
  subsample = 0.8, 
  colsample_bytree = 0.8
)

# Create 10-fold cross-validation
kfolds <- createFolds(y, k = 10, list = TRUE)

# Store results
cv_results <- sapply(kfolds, function(test_idx) {
  train_idx <- setdiff(seq_len(nrow(X)), test_idx)
  
  X_train <- X[train_idx, , drop = FALSE]
  y_train <- y[train_idx]
  X_test <- X[test_idx, , drop = FALSE]
  y_test <- y[test_idx]
  
  dtrain <- xgb.DMatrix(data = X_train, label = y_train)
  dtest <- xgb.DMatrix(data = X_test, label = y_test)
  
  # Train XGBoost model
  fit <- xgb.train(params = params, data = dtrain, nrounds = 300, verbose = 0)
  
  # Predictions
  preds <- predict(fit, dtest)
  
  # Compute RMSE
  return(RMSE(preds, y_test))
})

# Print average RMSE
mean_rmse <- mean(cv_results)
cat("Average XGBoost RMSE from Cross-Validation:", mean_rmse, "\n")
